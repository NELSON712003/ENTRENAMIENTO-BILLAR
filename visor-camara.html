// ============================================ // CONFIGURACIÓN // ============================================ const BROADCASTER_ID = 'mesa1-broadcaster-001'; // Debe coincidir con el ID del broadcaster const PEER_CONFIG = { host: '0.peerjs.com', port: 443, secure: true, debug: 1 }; // ============================================ // ELEMENTOS DOM // ============================================ const videoPlayer = document.getElementById('videoPlayer'); const waitingMessage = document.getElementById('waitingMessage'); const connectionStatus = document.getElementById('connectionStatus'); const connectionInfo = document.getElementById('connectionInfo'); const statusDot = document.getElementById('statusDot'); const statusText = document.getElementById('statusText'); const peerIdDisplay = document.getElementById('peerIdDisplay'); const qualityBadge = document.getElementById('qualityBadge'); // ============================================ // INICIALIZACIÓN PEERJS // ============================================ let peer = null; let reconnectAttempts = 0; const maxReconnectAttempts = 10; function init() { updateStatus('Conectando al servidor PeerJS...'); peer = new Peer(PEER_CONFIG); peer.on('open', (id) => { console.log('PeerJS conectado. Mi ID:', id); peerIdDisplay.textContent = `Viewer: ${id.substr(0, 8)}...`; connectionInfo.classList.remove('hidden'); connectToBroadcaster(); }); peer.on('error', (err) => { console.error('Error PeerJS:', err); handleError(err); }); peer.on('disconnected', () => { console.log('Desconectado del servidor, reconectando...'); updateStatus('Reconectando...'); statusDot.className = 'status-dot'; if (reconnectAttempts < maxReconnectAttempts) { reconnectAttempts++; setTimeout(() => { peer.reconnect(); }, 2000); } else { showError('No se pudo reconectar. Recarga la página.'); } }); } // ============================================ // CONEXIÓN AL BROADCASTER // ============================================ function connectToBroadcaster() { updateStatus(`Buscando transmisión: ${BROADCASTER_ID}...`); // Intentar llamar al broadcaster const call = peer.call(BROADCASTER_ID, null); if (!call) { // El broadcaster no está disponible aún, reintentar setTimeout(connectToBroadcaster, 2000); return; } handleIncomingCall(call); } function handleIncomingCall(call) { call.on('stream', (remoteStream) => { console.log('Stream recibido!'); waitingMessage.classList.add('hidden'); videoPlayer.classList.remove('hidden'); videoPlayer.srcObject = remoteStream; // Actualizar UI statusDot.className = 'status-dot connected'; statusText.textContent = 'Transmisión activa'; qualityBadge.textContent = 'LIVE'; qualityBadge.style.background = '#e74c3c'; // Detectar calidad del video detectVideoQuality(remoteStream); reconnectAttempts = 0; }); call.on('close', () => { console.log('Llamada cerrada'); handleDisconnect(); }); call.on('error', (err) => { console.error('Error en llamada:', err); handleError(err); }); } // ============================================ // MANEJO DE EVENTOS // ============================================ function handleDisconnect() { videoPlayer.classList.add('hidden'); waitingMessage.classList.remove('hidden'); updateStatus('Transmisión finalizada. Reconectando...'); statusDot.className = 'status-dot'; // Reintentar conexión setTimeout(connectToBroadcaster, 3000); } function handleError(err) { let message = 'Error de conexión'; if (err.type === 'peer-unavailable') { message = 'La mesa no está transmitiendo aún. Esperando...'; // Reintentar en 3 segundos setTimeout(connectToBroadcaster, 3000); } else if (err.type === 'network') { message = 'Error de red. Verifica tu conexión.'; } else if (err.type === 'server-error') { message = 'Error del servidor. Reintentando...'; setTimeout(init, 5000); } updateStatus(message); statusDot.className = 'status-dot error'; statusText.textContent = 'Error'; } function showError(msg) { waitingMessage.innerHTML = `
⚠️
Error
${msg}
`; } // ============================================ // UTILIDADES // ============================================ function updateStatus(msg) { connectionStatus.textContent = msg; console.log('[Status]', msg); } function detectVideoQuality(stream) { const videoTrack = stream.getVideoTracks()[0]; if (videoTrack) { const settings = videoTrack.getSettings(); const width = settings.width || 0; if (width >= 1920) qualityBadge.textContent = 'FHD'; else if (width >= 1280) qualityBadge.textContent = 'HD'; else if (width >= 854) qualityBadge.textContent = '480p'; else qualityBadge.textContent = 'SD'; } } function toggleFullscreen() { if (!document.fullscreenElement) { document.documentElement.requestFullscreen().catch(e => { console.log('Fullscreen no soportado'); }); } else { document.exitFullscreen(); } } // Reconectar cuando el video se pause (pérdida de conexión) videoPlayer.addEventListener('pause', () => { if (!videoPlayer.ended && videoPlayer.srcObject) { console.log('Video pausado inesperadamente, reconectando...'); setTimeout(connectToBroadcaster, 2000); } }); // Iniciar init();
